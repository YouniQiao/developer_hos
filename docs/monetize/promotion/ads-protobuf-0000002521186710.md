# protoBuf接口协议声明

![](./img/note_3.0-zh-cn_1865f8c401c7.png) 

声明：该协议将于2026年3月1日上线。

```
syntax="proto2";
message Size {
optional int32 width = 1;
optional int32 height = 2;
}
message CreativeInfo {
optional int64 creativeId = 1;  // 创意ID，如果是创意2.0是creative_id，如果是创意3.0是元素组id
}
message PreciseAdId {
optional int64 adgroupId = 1;   // 任务ID
repeated CreativeInfo creativeIds = 3;
}
message DpaInfo {
optional int64 productLibId = 1;    // DPA商品库ID
repeated string dpaProductId = 2;
}
message AdInfo {
optional string userId = 1;
optional string rtaId = 2;
repeated PreciseAdId adIds = 3;
repeated DpaInfo dpaInfos = 4;  // 商品维度出价不需要用到任务和创意会议信息
}
message RtaReq {
optional string requestId = 1; // 请求唯一ID
optional int64 requestTime = 2; // 请求时间戳
optional string channelName = 3; // 渠道信息，填 HuaweiAds
repeated string rtaIdList = 4; // rtaIdList, 每个rtaId对应多个账户id
optional int32 publishType = 5; // 站点类型，0 内部站点，1 外部站点
optional string oaid = 6; // 设备oaid，oaid和dfa三个字段至少有一个不为空
optional string ifa = 7; // ios系统表示idfa, Android系统表示gaid、oaid、dfa三个字段至少有一个不为空
repeated string userList = 8; // 账户id，一个rtaId下可以多个账户
optional string sign = 9; // 媒体行业分类
optional string androidVersion = 10; // 安卓版本号
optional string emuiVersion = 11; // EMUI版本
optional string transUniqueId = 13; // 广告位关联ID
optional string propagationName = 15; // 机型
repeated int64 creativeSizeIds = 16; // 版位ID
optional string ip = 17; // IP地址
optional string ua = 18; // 浏览器UserAgent
repeated Size slotSize = 19; // 广告位尺寸
optional int32 adType = 20; // 广告类型
optional string embv = 29; // embedding向量,用户表征向量字符串（BASE64编码），有OAID时才发送
repeated RtaShareTag rtaShareTag = 30;
repeated AppOpenInfo appOpenList = 31;
repeated AdInfo adInfos = 32;  // 精排二次请求需要调价的ID集合
}
message AppOpenInfo {
optional string appPkg = 1;
repeated string appOpenNames = 2;
}
message RtaShareTag{
optional string labelKey = 1;
repeated string packageNames = 2;  // 包名
}
message Product {
optional string productId = 1; // 商品ID
optional int32 score = 2; // 商品置信度打分:0-4分，4分表示推荐可靠度高
}
message PriceInfo {
optional string currency = 1; // 可指定币种，不传默认使用任务币种
optional double cpm = 2;
optional double cpc = 3;
optional double cpd = 4;
optional double cpi = 5;
optional double cpa = 6;
}
message OcpcPriceInfo {
optional int32 convType = 1; // 目标转化类型
optional string currency = 2; // 可指定币种，不传默认使用任务币种
optional double price = 3; // 出价
optional double priceWeight = 4; // 出价系数
optional int32 deepConvType = 5; // 双出价的深层目标转化类型。只有双出价时才生效，默认未配置时是单出价。配置双出价时需要先配置浅层转化目标convType。
}
message UserInfo {
repeated string userIdList = 1; // 账号ID
repeated string rtaIdList = 2; // 商品库ID
optional string platformId = 3; // 商品库ID
repeated string productList = 4; // 多个商品ID信息
optional double priceWeight = 5; // 出价比例
repeated int64 creativeIds = 6; // 指定的创意ID
optional PriceInfo price = 7; // 根据计费方式返回不同出价
optional int32 recMode = 8; // 1: LBS推荐 0: 行为库推荐
repeated Product productExtList = 9; // 带分值的商品列表
repeated OcpcPriceInfo ocpcPriceInfoList = 10; // oCPC任务不同目标转化类型的出价
}
message RtaResponse {
optional int32 retCode = 1; // 返回码,正常返回0，其他表示异常
optional string errorMsg = 2; // 返回的错误消息
optional string result = 3; // 生效类型，取值yes/no,yes表示账户全部生效，no表示账户全部不生效
repeated string rtaIdList = 4;
repeated string userList = 5;
optional double score = 6;
repeated UserInfo userInfoList = 7;
repeated string expId = 8;   // 实验id
repeated CreatePriceInfo creativePriceInfos = 9;
repeated DpaPriceInfo dpaPriceInfos = 10;
}
message CreatePriceInfo{
repeated int64 ids = 1;
optional double priceWeight = 2;
optional double price = 3;
repeated OcpcPriceInfo ocpcPriceInfoList = 4;
}
message DpaPriceInfo{
optional int64 productLibId = 1;
repeated string productIds = 2;
optional double priceWeight = 3;
optional double price = 4;
optional string currency = 5;
repeated OcpcPriceInfo ocpcPriceInfoList = 7;
}
```
