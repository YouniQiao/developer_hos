对于自定义事件，开发者除了需要注册对应的事件处理器，还需要手动发布事件来触发事件处理器逻辑。

```
using Pgd.Event;

// 1) 定义自定义事件，需继承自IEvent接口
public readonly struct PlayerScoredEvent : IEvent
{
    public readonly int PlayerId;
    public readonly int Score;
    public PlayerScoredEvent(int playerId, int score) { PlayerId = playerId; Score = score; }
}

// 2) 开启EventHub
world.EnableEventHub();

// 3) 注册事件处理器
world.RegisterHandler<PlayerScoredEvent>(e => Debug.Log($"Player {e.PlayerId} +{e.Score}"));

// 4) 发布事件
world.Publish(new PlayerScoredEvent(1, 5));
world.Publish(new PlayerScoredEvent(1, 20));

// 5) 查询/管理
int cnt = world.GetHandlerCount<PlayerScoredEvent>(); // 获取PlayerScoredEvent注册的事件处理器数量
bool has = world.HasHandlers<PlayerScoredEvent>(); // 判断PlayerScoredEvent是否注册了事件处理器
int totalCnt = world.GetHandlersCount(); // 获取所有事件注册的事件处理器总和
int eventCnt = world.GetEventCount(); // 获取注册了事件处理器的事件总和
bool isEnabled = world.IsEventHubEnabled; // 查询EventHub是否启用的属性
```
