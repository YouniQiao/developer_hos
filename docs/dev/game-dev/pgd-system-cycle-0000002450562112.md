public class GameStateSystem : PgdSystem<GameState>
{
    protected override void OnAddWorld(IECSWorld world)
    {
        Console.WriteLine($"系统添加到world: {world}");

        // 初始化系统特定的资源
        InitializeForWorld(world);
    }

    protected override void OnRemoveWorld(IECSWorld world)
    {
        Console.WriteLine($"系统从world移除: {world}");

        // 清理系统特定的资源
        CleanupForWorld(world);
    }

    public override void OnUpdateCollectionBegin()
    {
        // 在系统集合开始更新前调用
        PrepareFrameData();
    }

    protected override void OnUpdate()
    {
        // 核心更新逻辑
        GetQuery().ForEachEntity((ref GameState state, IEntity entity) => {
            UpdateGameState(ref state);
        });
    }

    public override void OnUpdateCollectionEnd()
    {
        // 在系统集合更新完成后调用
        FinalizeFrameData();
    }

    private void InitializeForWorld(IECSWorld world) { /* 初始化逻辑 */ }
    private void CleanupForWorld(IECSWorld world) { /* 清理逻辑 */ }
    private void PrepareFrameData() { /* 准备帧数据 */ }
    private void UpdateGameState(ref GameState state) { /* 更新游戏状态 */ }
    private void FinalizeFrameData() { /* 完成帧数据 */ }
}
