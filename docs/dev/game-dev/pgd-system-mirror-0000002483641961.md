## System调试信息

```
// 获取系统基本信息
Console.WriteLine($"系统ID: {movementSystem.Id}");
Console.WriteLine($"系统名称: {movementSystem.SystemName}");
Console.WriteLine($"系统激活: {movementSystem.Activated}");
Console.WriteLine($"父集合: {movementSystem.ParentCollection}");

// 查询系统特有信息
if (movementSystem is QueryPgdSystemBase querySystem)
{
    Console.WriteLine($"实体数量: {querySystem.GetQuery().EntityCount}");
    Console.WriteLine($"查询组件: {querySystem.GetQuery().Components}");
    Console.WriteLine($"查询: {querySystem.GetQuery()}");
}

// 系统集合调试信息
Console.WriteLine($"子系统数量: {world.PgdSystemManager.ChildSystems.Count}");
foreach (var childSystem in world.PgdSystemManager.ChildSystems)
{
    Console.WriteLine($"- {childSystem}");
}
```
