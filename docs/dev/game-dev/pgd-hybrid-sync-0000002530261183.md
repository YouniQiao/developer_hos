开发者可自定义数据同步System，使用GoLink实现GameObject和Entity之间的数据同步。如下以Transform属性为例：

```
public class HybridTransformSync : PgdSystem<GoLink, PgdLocalTransform>
{
    protected override void OnUpdate()
    {
        GetQuery().ForEachEntity((
            ref GoLink goLink, ref PgdLocalTransform transform, IEntity entity) =>
        {
            if (goLink.transform == null || goLink.gameObject == null)
            {
                Debug.LogError($"PgdHybridTransformSync: Invalid GoLink, gameObject or transform is null, entityId: {entity.Id}");
                return;
            }

            goLink.transform.position = transform.Position;
            goLink.transform.rotation = transform.Rotation;
        });
    }
}
```
