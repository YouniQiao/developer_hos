## 创建系统集合

```
// 创建系统集合来组织相关系统
var renderingCollection = new PgdSystemCollection("Rendering");
renderingCollection.Add(new CameraSystem());
renderingCollection.Add(new LightingSystem());
renderingCollection.Add(new MeshRenderSystem());
renderingCollection.Add(new UIRenderSystem());

var physicsCollection = new PgdSystemCollection("Physics");
physicsCollection.Add(new CollisionDetectionSystem());
physicsCollection.Add(new RigidBodySystem());
physicsCollection.Add(new TriggerSystem());

var gameplayCollection = new PgdSystemCollection("Gameplay");
gameplayCollection.Add(new PlayerInputSystem());
gameplayCollection.Add(new AISystem());
gameplayCollection.Add(new WeaponSystem());
gameplayCollection.Add(new InventorySystem());

// 添加到主系统管理器
world.RegisterSystem(physicsCollection);
world.RegisterSystem(gameplayCollection);
world.RegisterSystem(renderingCollection);

// 更新所有系统
world.Update(default);

// 更新physicsCollection
physicsCollection.Update(default);

// 更新gameplayCollection
gameplayCollection.Update(default);

// 更新renderingCollection
renderingCollection.Update(default);
```

## 嵌套系统集合

```
// 创建嵌套的系统集合
var mainGameplay = new PgdSystemCollection("MainGameplay");

var playerSystems = new PgdSystemCollection("PlayerSystems");
playerSystems.Add(new PlayerMovementSystem());
playerSystems.Add(new PlayerCombatSystem());
playerSystems.Add(new PlayerInventorySystem());

var enemySystems = new PgdSystemCollection("EnemySystems");
enemySystems.Add(new EnemyAISystem());
enemySystems.Add(new EnemySpawnerSystem());
enemySystems.Add(new EnemyCombatSystem());

// 构建层次结构
mainGameplay.Add(playerSystems);
mainGameplay.Add(enemySystems);
world.RegisterSystem(mainGameplay);

Console.WriteLine(world.PgdSystemManager);
// 输出层次结构信息
```

## 系统查找和管理

```
// 首先构建一个复杂的嵌套系统结构来演示recursive的作用
var gameplayCollection = new PgdSystemCollection("Gameplay");
var playerSystems = new PgdSystemCollection("PlayerSystems");
playerSystems.Add(new PlayerMovementSystem());
playerSystems.Add(new PlayerCombatSystem());

var enemySystems = new PgdSystemCollection("EnemySystems");
enemySystems.Add(new EnemyAISystem());

gameplayCollection.Add(playerSystems);
gameplayCollection.Add(enemySystems);
world.RegisterSystem(gameplayCollection);

/*
系统结构：
Root (IECSWorld.PgdSystemManager)
└── Gameplay (PgdSystemCollection)
    ├── PlayerSystems (PgdSystemCollection)
    │   ├── PlayerMovementSystem
    │   └── PlayerCombatSystem
    └── EnemySystems (PgdSystemCollection)
        └── EnemyAISystem
*/

// === recursive = false (非递归查找) ===
// 只在直接子系统中查找，不会深入嵌套集合

// 在Root中查找直接子集合 - 可以找到
var gameplayDirect = world.FindCollection("Gameplay", recursive: false);
Console.WriteLine($"直接查找Gameplay: {gameplayDirect != null}"); // True

// 在Root中查找嵌套子集合 - 找不到
var playerSystemsDirect = world.FindCollection("PlayerSystems", recursive: false);
Console.WriteLine($"直接查找PlayerSystems: {playerSystemsDirect != null}"); // False

// 在Root中查找嵌套系统 - 找不到
var playerMovementDirect = world.FindSystem<PlayerMovementSystem>(recursive: false);
Console.WriteLine($"直接查找PlayerMovementSystem: {playerMovementDirect != null}"); // False

// === recursive = true (递归查找) ===
// 会深入整个系统树查找，包括所有嵌套层级

// 在Root中递归查找嵌套子集合 - 可以找到
var playerSystemsRecursive = world.FindCollection("PlayerSystems", recursive: true);
Console.WriteLine($"递归查找PlayerSystems: {playerSystemsRecursive != null}"); // True

// 在Root中递归查找深层嵌套的系统 - 可以找到
var playerMovementRecursive = world.FindSystem<PlayerMovementSystem>(recursive: true);
Console.WriteLine($"递归查找PlayerMovementSystem: {playerMovementRecursive != null}"); // True

var enemyAI = world.FindSystem<EnemyAISystem>(recursive: true);
Console.WriteLine($"递归查找EnemyAISystem: {enemyAI != null}"); // True

// === 实际应用场景 ===

// 场景1：快速查找根级系统（推荐使用 recursive: false）
var rootLevelSystems = world.FindCollection("Gameplay", recursive: false);
// 只查找直接子系统，性能更好

// 场景2：全局系统查找（使用 recursive: true）
var anywhereSystem = world.FindSystem<PlayerMovementSystem>(recursive: true);
// 不管系统在哪个层级，都能找到

// 场景3：在特定集合中查找（根据情况选择）
var playerMovementInEnableSystemCollection = gameplayCollection.FindSystem<PlayerMovementSystem>(recursive: true);
// 在Gameplay组中递归查找PlayerMovementSystem

var directChildOnly = gameplayCollection.FindCollection("PlayerSystems", recursive: false);
// 只查找Gameplay的直接子集合

// 系统移动
if (playerMovementRecursive != null)
{
    // 将玩家移动系统移动到不同的集合
    var newIndex = playerMovementRecursive.MoveToCollection(enemySystems, index: 0);
    Console.WriteLine($"系统移动到索引: {newIndex}");
}

// 系统激活/禁用
var renderingCollection = world.FindCollection("Rendering", recursive: true);
if (renderingCollection != null)
{
    renderingCollection.Activated = false; // 禁用整个渲染组
    world.Update(tick); // 渲染组不会被更新

    renderingCollection.Activated = true; // 重新启用
}
```

**recursive参数详解**

| 参数值 | 查找范围 | 性能 | 适用场景 |
| --- | --- | --- | --- |
| recursive: false | 仅直接子系统 | 更快 | 查找已知层级的系统，扁平结构。 |
| recursive: true | 整个系统树 | 较慢 | 不确定系统位置，复杂嵌套结构。 |

```
public class SystemLookupExample
{
    private readonly IECSWorld world;

    public void PerformanceDemonstration()
    {
        // 推荐：已知系统在根级别时使用非递归查找
        var rootSystem = world.FindCollection("Gameplay", recursive: false);

        // 必要：需要全局查找时使用递归查找
        var deepSystem = world.FindSystem<SomeDeepNestedSystem>(recursive: true);

        // 优化：在具体集合中查找，减少搜索范围
        var gameplayCollection = world.FindCollection("Gameplay", recursive: false);
        if (gameplayCollection != null)
        {
            // 在已知集合中查找，比全局递归查找更高效
            var playerSystem = gameplayCollection.FindSystem<PlayerMovementSystem>(recursive: true);
        }
    }
}
```
