## 标签过滤

```
// 查询必须包含所有指定标签的实体
var playerQuery = world.Query<Health>()
    .WithAllTags(ITags.Get<PlayerTag, AliveTag>());

// 查询包含任意指定标签的实体
var anyEnemyQuery = world.Query<Health>()
    .WithAnyTags(ITags.Get<EnemyTag, BossTag>());

// 查询不包含特定标签的实体
var nonPlayerQuery = world.Query<Health>()
    .WithoutAllTags(ITags.Get<PlayerTag>());

// 查询不包含任意指定标签的实体
var noneTagQuery = world.Query<Health>()
    .WithoutAnyTags(ITags.Get<DeadTag, DisabledTag>());
```

## 组件过滤

```
// 查询必须包含额外组件的实体
var weaponizedQuery = world.Query<Health>()
    .WithAllComponents(IComponents.Get<Weapon, Inventory>());

// 查询包含任意额外组件的实体
var anySpecialQuery = world.Query<Health>()
    .WithAnyComponents(IComponents.Get<Magic, Technology>());

// 查询不包含特定组件的实体
var basicQuery = world.Query<Health>()
    .WithoutAllComponents(IComponents.Get<Weapon, Inventory>());

// 查询不包含任意指定组件的实体
var noneComponentQuery = world.Query<Health>()
    .WithoutAnyComponents(IComponents.Get<Weapon, Inventory>());
```

## 复杂过滤组合

```
// 链式调用构建复杂查询
var complexQuery = world.Query<Health, PgdPosition>()
    .WithAllTags(ITags.Get<AliveTag>())           // 必须活着
    .WithAnyTags(ITags.Get<PlayerTag, NPCTag>())  // 是玩家或NPC
    .WithoutAllTags(ITags.Get<FrozenTag>())       // 不能被冰冻
    .WithAllComponents(IComponents.Get<Velocity>()) // 必须能移动
    .WithoutAnyComponents(IComponents.Get<Stunned>()); // 不能被眩晕

complexQuery.ForEachEntity((ref Health health, ref PgdPosition position, IEntity entity) => {
    // 处理符合条件的实体
});
```

## 包含非激活实体

```
// 默认查询只包含激活的实体
var activeQuery = world.Query<Health>();
Console.WriteLine($"激活实体: {activeQuery.EntityCount}");

// 包含非激活实体
var allQuery = world.Query<Health>().WithInactive();
Console.WriteLine($"所有实体: {allQuery.EntityCount}");
```

## 查询冻结

```
// 冻结查询过滤器，防止意外修改
var frozenQuery = world.Query<Health>()
    .WithAllTags(ITags.Get<PlayerTag>())
    .FreezeFilter(); // 冻结后不能再修改过滤条件

// 尝试修改冻结的查询会抛出异常
try
{
    frozenQuery.WithAnyTags(ITags.Get<EnemyTag>());
}
catch (InvalidOperationException ex)
{
    Console.WriteLine("不能修改已冻结的查询过滤器");
}
```
