## 完整的游戏系统设置

```
public class GameSystemManager
{
    private readonly IECSWorld world;

    public GameSystemManager()
    {
        world = new IECSWorld();

        SetupSystems();
    }

    private void SetupSystems()
    {
        // 输入系统 - 最高优先级
        var inputCollection = new PgdSystemCollection("Input");
        inputCollection.Add(new KeyboardInputSystem());
        inputCollection.Add(new MouseInputSystem());
        inputCollection.Add(new GamepadInputSystem());

        // 游戏逻辑系统
        var gameplayCollection = new PgdSystemCollection("Gameplay");
        gameplayCollection.Add(new PlayerControllerSystem());
        gameplayCollection.Add(new AIBehaviorSystem());
        gameplayCollection.Add(new WeaponSystem());
        gameplayCollection.Add(new InventorySystem());
        gameplayCollection.Add(new QuestSystem());

        // 物理系统
        var physicsCollection = new PgdSystemCollection("Physics");
        physicsCollection.Add(new MovementSystem());
        physicsCollection.Add(new CollisionSystem());
        physicsCollection.Add(new TriggerSystem());

        // 渲染系统 - 最低优先级
        var renderCollection = new PgdSystemCollection("Rendering");
        renderCollection.Add(new CameraSystem());
        renderCollection.Add(new SpriteRenderSystem());
        renderCollection.Add(new ParticleSystem());
        renderCollection.Add(new UISystem());

        // 按执行顺序添加系统组
        world.RegisterSystem(inputCollection);
        world.RegisterSystem(gameplayCollection);
        world.RegisterSystem(physicsCollection);
        world.RegisterSystem(renderCollection);
    }

    public void Update(float deltaTime, float totalTime)
    {
        var tick = new UpdateTick(deltaTime, totalTime);
        world.Update(tick);
    }

    public T GetSystem<T>() where T : PgdSystemBase
    {
        return world.FindSystem<T>(recursive: true);
    }

    public void EnableSystemCollection(string EnableSystemCollectionName, bool enabled)
    {
        var EnableSystemCollection = world.FindCollection(EnableSystemCollectionName, recursive: true);
        if (EnableSystemCollection != null)
        {
            EnableSystemCollection.Activated = enabled;
        }
    }
}
```

## 条件执行和动态系统管理

```
public class AdaptiveSystemManager : PgdSystemBase
{
    private bool isInMenus = false;
    private bool isPaused = false;
    private PgdSystemCollection gameplaySystems;
    private PgdSystemCollection renderSystems;

    protected override void OnAddWorld(IECSWorld world)
    {
        gameplaySystems = ParentCollection.FindCollection("Gameplay", false);
        renderSystems = ParentCollection.FindCollection("Rendering", false);
    }

    protected override void OnUpdateCollection()
    {
        // 根据游戏状态动态启用/禁用系统
        UpdateSystemStates();
    }

    private void UpdateSystemStates()
    {
        // 在菜单中时禁用游戏逻辑
        if (gameplaySystems != null)
        {
            gameplaySystems.Activated = !isInMenus;
        }

        // 暂停时禁用除UI外的所有系统
        if (isPaused)
        {
            var uiSystem = renderSystems?.FindSystem<UISystem>(false);
            if (renderSystems != null)
            {
                foreach (var system in renderSystems.ChildSystems)
                {
                    system.Activated = system == uiSystem;
                }
            }
        }
    }

    public void SetGameState(bool inMenus, bool paused)
    {
        isInMenus = inMenus;
        isPaused = paused;
    }
}
```
