## 泛型查询系统

PGD提供了1~10个组件的泛型查询系统。

查询系统会自动排除带有Inactive标签的实体。若查询Inactive标签，需手动设置过滤。

```
// 无组件约束的系统
public class GlobalSystem : PgdSystem
{
    protected override void OnUpdate()
    {
        // 处理全局逻辑，查询所有实体
        var query = GetQuery();
        Console.WriteLine($"总实体数: {query.EntityCount}");
    }

    protected override void OnAddWorld(IECSWorld world)
    {
        base.OnAddWorld(world);
        GetQuery().WithInactive(); // 通过GetQuery调用WithInactive()接口使能System对Inactive实体的查询
    }
}

// 单组件系统
public class HealthRegenSystem : PgdSystem<Health>
{
    protected override void OnUpdate()
    {
        GetQuery().ForEachEntity((ref Health health, IEntity entity) => {
            if (health.current < health.maximum)
            {
                health.current = Math.Min(health.current + 5.0f * Tick.DeltaTime, health.maximum);
            }
        });
    }
}

// 多组件系统
public class PhysicsSystem : PgdSystem<PgdPosition, Velocity, Mass>
{
    protected override void OnUpdate()
    {
        GetQuery().ForEachEntity((ref PgdPosition position, ref Velocity velocity, ref Mass mass, IEntity entity) => {
            // 应用重力
            velocity.value.Y -= 9.8f * Tick.DeltaTime;

            // 更新位置
            position.vec3 += velocity.value * Tick.DeltaTime;

            // 应用阻力
            velocity.value *= 1.0f - (0.1f * Tick.DeltaTime);
        });
    }
}

// 最多支持10个组件
public class ComplexGameplaySystem : PgdSystem<Health, PgdPosition, Velocity, Weapon, Inventory, Stats, AI, Buff>
{
    protected override void OnUpdate()
    {
        GetQuery().ForEachEntity((ref Health health, ref PgdPosition pos, ref Velocity vel,
                                 ref Weapon weapon, ref Inventory inv, ref Stats stats,
                                 ref AI ai, ref Buff buff, IEntity entity) => {
            // 复杂的游戏逻辑处理
        });
    }
}
```

## 查询过滤器

```
public class PlayerSystem : PgdSystem<Health, PgdPosition>
{
    protected override void OnAddWorld(IECSWorld world)
    {
        // 在系统初始化时配置查询过滤器
        QueryFilter
            .WithAllTags(ITags.Get<PlayerTag, AliveTag>())
            .WithoutAllTags(ITags.Get<DisabledTag>())
            .WithAllComponents(IComponents.Get<Velocity>());
    }

    protected override void OnUpdate()
    {
        // 只处理符合过滤条件的实体
        GetQuery().ForEachEntity((ref Health health, ref PgdPosition position, IEntity entity) => {
            // 只有活着的、启用的、有速度组件的玩家实体会被处理
        });
    }
}
```

## 高性能ArchetypeChunk迭代

```
public class OptimizedMovementSystem : PgdSystem<PgdPosition, Velocity>
{
    protected override void OnUpdate()
    {
        // 使用ArchetypeChunk获得最佳性能
        foreach (var (positionSegment, velocitySegment, entities) in GetQuery().ArchetypeChunk)
        {
            var positions = positionSegment.Span;
            var velocities = velocitySegment.Span;

            // 直接操作内存数组，获得最佳性能
            for (int i = 0; i < positionSegment.Length; i++)
            {
                positions[i].vec3 += velocities[i].value * Tick.DeltaTime;
            }
        }
    }
}
```
