## 场景介绍

图片比例裁剪是拍摄美化类应用的典型场景之一，如编辑图片时，用户可选择不同的裁剪比例（1：1、4：3、16：9），并通过拖拽裁剪框调节图片大小，从而提升图片的视觉效果。

本示例通过滑动手势[PanGesture](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/ts-basic-gestures-pangesture)和[PixelMap.crop](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-image-pixelmap#crop9)方法实现图片的裁剪功能，并使用[PhotoAccessHelper](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-photoaccesshelper-photoaccesshelper)将裁剪后的图片保存到相册中。

1. 获取图片的信息，绘制裁剪框的初始位置。

   ```
   @Component
   export struct CropView {
     @State isScrollingRect: boolean = false;
     @ObjectLink cropRect: CropRect;
     @Prop cropRectClone: CropRect;
     @Prop scaleRatio: number;
     @Consume('pixelMap') pixelMap: image.PixelMap;
     @Consume('isPixelMapChange') isPixelMapChange: boolean;
     onCropComplete?: (newPixelMap: image.PixelMap) => void;
     private dragObj: DragObj = new DragObj(false);
     // 可配置的边距比例（裁剪框最大宽度占容器的比例）
     private static readonly MAX_WIDTH_RATIO: number = 0.85;
     private containerWidth: number = 0;

     aboutToAppear() {
       this.getContainerWidth();
     }

     private getContainerWidth(): void {
       try {
         // 获取CropView所在容器的宽度
         const componentInfo = this.getUIContext().getComponentUtils().getRectangleById('crop_view_container');
         this.containerWidth = this.getUIContext().px2vp(componentInfo.size.width);
         hilog.info(0x0000, TAG, `containerWidth: ${this.containerWidth}`);
       } catch (error) {
         this.containerWidth = 300;
       }
     }

     private getMaxAllowedWidth(): number {
       return this.containerWidth * CropView.MAX_WIDTH_RATIO;
     }

     build() {
       Column() {
         Column() {
           // 8个操作区
           Blank().operCorner({ top: 3, left: 3 }).position({ x: -4, y: -4 });
           Blank().operCorner({ top: 3 }).position({ x: this.cropRect.width / 2 - 10, y: -4 });
           Blank().operCorner({ top: 3, right: 3 }).position({ x: this.cropRect.width - 16, y: -4 });
           Blank().operCorner({ left: 3 }).position({ x: -4, y: this.cropRect.height / 2 - 10 });
          // ...
         }
         .border({ color: 0xAFFFFFFF, width: 1 })
         .width(this.cropRect.width)
         .height(this.cropRect.height)
         .position({ x: Constants.CLIP_HOT_ZONE_POSITION, y: Constants.CLIP_HOT_ZONE_POSITION });
       }
       .id('crop_view_container')
       .width(this.cropRect.width + Constants.CLIP_HOT_ZONE_WIDTH)
       .height(this.cropRect.height + Constants.CLIP_HOT_ZONE_WIDTH)
       .position({
         x: this.cropRect.x - Constants.CLIP_HOT_ZONE_POSITION,
         y: this.cropRect.y - Constants.CLIP_HOT_ZONE_POSITION
       })
       // ...
     }
     // ...
   }
   ```
2. 给裁剪框添加滑动手势监听，设置裁剪热区。

   ```
   // 给裁剪框添加滑动手势监听
   // ...
   .gesture(
     PanGesture({ distance: 0.01, fingers: 1 })
       .onActionStart((event: GestureEvent) => {
         // 每次开始拖拽时更新容器宽度（确保最新）
         this.getContainerWidth();
         if (this.isCropFrameHotArea(event)) {
           hilog.info(0x0000, TAG, 'start drag frame');
         }
       })
       .onActionUpdate((event: GestureEvent) => {
         if (!event || !this.dragObj.dragging) {
           return;
         }
         this.getSingleCropRect(event.offsetX, event.offsetY,
           new CropRect(this.cropRect.x, this.cropRect.y, this.cropRect.width, this.cropRect.height),
           this.dragObj.action);
       })
       .onActionEnd(async () => {
         if (!this.dragObj.dragging) {
           return;
         }

         this.isScrollingRect = false;
         this.dragObj.dragging = false;

         const newPixelMap = await cropImage(
           this.pixelMap,
           this.cropRect,
           this.cropRectClone,
           this.scaleRatio
         );

         if (newPixelMap && this.onCropComplete) {
           this.onCropComplete(newPixelMap);
         }
         this.isPixelMapChange = !this.isPixelMapChange;
       })
     );
   ```

   ```
   // 设置裁剪热区
   isCropFrameHotArea(event: GestureEvent): boolean {
     const fingerInfos = event.fingerList;
     if (!event || fingerInfos.length !== 1) {
       return false;
     }

     // 使用local坐标（相对于CropView组件）
     const localX = fingerInfos[0].localX;
     const localY = fingerInfos[0].localY;

     const offset = Constants.CLIP_HOT_ZONE_POSITION;
     const rectLeft = offset;
     const rectRight = offset + this.cropRect.width;
     const rectTop = offset;
     const rectBottom = offset + this.cropRect.height;

     const HOT_ZONE = 20;

     const onLeft = Math.abs(localX - rectLeft) <= HOT_ZONE;
     const onRight = Math.abs(localX - rectRight) <= HOT_ZONE;
     const onTop = Math.abs(localY - rectTop) <= HOT_ZONE;
     const onBottom = Math.abs(localY - rectBottom) <= HOT_ZONE;

     if (onLeft || onRight || onTop || onBottom) {
       this.isScrollingRect = true;
       const cacheRect = this.cropRect.clone();
       const action = new Action(onLeft, onTop, onRight, onBottom);
       this.dragObj = new DragObj(true, fingerInfos[0].globalX, fingerInfos[0].globalY, action, cacheRect);
       return true;
     }
     return false;
   }
   ```

## 约束与限制

* 本示例支持API Version 20 Release及以上版本。
* 本示例支持HarmonyOS 6.0.0 Release SDK及以上版本。
* 本示例需要使用DevEco Studio 6.0.0 Release及以上版本进行编译运行。
