## 问题现象

跳转到视频播放页面时，标签栏依然存在，影响用户体验。

![](./img/0ede3293.png "点击放大")

## 背景知识

* [Web](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-basic-components-web)组件提供网页显示的能力。
* 设置[onLoadIntercept](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-basic-components-web-events#onloadintercept10)时，当Web组件加载url之前触发该回调，用于判断是否阻止此次访问。
* H5页面可通过设置window.location.href进行H5页面跳转。
* [Navigation](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/ts-basic-components-navigation)组件是路由导航的根视图容器，一般作为Page页面的根容器使用，其内部默认包含了标题栏、内容区和工具栏，其中内容区默认首页显示导航内容（Navigation的子组件）或非首页显示（[NavDestination](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/ts-basic-components-navdestination)的子组件），首页和非首页通过路由进行切换。

## 问题定位

查看跳转前后显示页面的Web组件，跳转前后的页面都是通过同一个Tabs组件下的Web组件显示。通常情况下，跳转到某些特殊页面，如视频播放页面时，应使用不在Tabs组件内的Web组件显示。

```
Tabs({ barPosition: BarPosition.End, index: this.currentIndex, controller: this.tabscontroller }) {
  TabContent() {
    Column() {
      // 加载Web页面
      Web({ src: $rawfile('button.html'), controller: this.controller })
        .height('100%')
        .width('100%')
    }
    .width('100%').height('100%')
  }.tabBar(this.tabBuilder(0, '标签栏1'))

  TabContent() {
    Column().width('100%').height('100%')
  }.tabBar(this.tabBuilder(1, '标签栏2'))

  TabContent() {
    Column().width('100%').height('100%')
  }.tabBar(this.tabBuilder(2, '标签栏3'))
}
```

## 分析结论

跳转前后的页面都是通过同一个Tabs组件下的Web组件显示，导致视频播放页面出现标签栏。

## 修改建议

Web组件使用onLoadIntercept和Navigation+NavDestination，跳转到视频播放页面时使用非Tabs组件下的Web组件显示页面。

Index.ets:

```
import { webview } from '@kit.ArkWeb';

@Entry
@Component
struct Index {
  pageInfos: NavPathStack = new NavPathStack();
  fontColor: string = '#182431';
  selectedFontColor: string = '#007DFF';
  @State currentIndex: number = 0;
  @State selectedIndex: number = 0;
  marginLeft: number = 30;
  private tabsController: TabsController = new TabsController();
  color: string = '#DB7093';
  controller: webview.WebviewController = new webview.WebviewController();

  @Builder
  tabBuilder(index: number, name: string) {
    Column() {
      Image($r('app.media.startIcon'))
        .width(40)
        .height(40)
        .objectFit(ImageFit.Contain)
        .margin({ top: 17})
      Text(name)
        .fontColor(this.selectedIndex === index ? this.selectedFontColor :
          this.fontColor)
        .fontSize(16)
        .fontWeight(this.selectedIndex === index ? 500 : 400)
        .lineHeight(22)
        .margin({ top: 10, bottom: 7 })
        .textAlign(TextAlign.Center)
        .width('100%')
    }
    .width('30%')
    .alignItems(HorizontalAlign.Center)
  }

  build() {
    Navigation(this.pageInfos) {
      Stack() {
        Tabs({ barPosition: BarPosition.End, index: this.currentIndex, controller: this.tabsController }) {
          TabContent() {
            Column() {
              // 加载Web页面
              Web({ src: $rawfile('button.html'), controller: this.controller })
                .height('100%')
                .width('100%')
                .fileAccess(true)
                .geolocationAccess(false)
                // 加载新页面前获取新页面地址
                .onLoadIntercept((event) => {
                  let url = event.data.getRequestUrl();
                  console.info('url is ' + url);
                  // 跳转到视频播放页面时跳转到新的Web组件播放
                  if (url === 'resource://rawfile/index_cn.html') {
                    this.pageInfos.pushPath({ name: 'newWeb', param: 'index_cn.html' });
                    return true; // 当前Web组件不加载该页面
                  }
                  return false;
                });
            }
            .width('100%').height('100%');
          }.tabBar(this.tabBuilder(0, '标签栏1'));

          TabContent() {
            Column().width('100%').height('100%');
          }.tabBar(this.tabBuilder(1, '标签栏2'));

          TabContent() {
            Column().width('100%').height('100%');
          }.tabBar(this.tabBuilder(2, '标签栏3'));
        }
        .barMode(BarMode.Scrollable)
        .barWidth('100%')
        .barHeight(100)
        .onChange((index: number) => {
          // currentIndex控制TabContent显示页签
          this.currentIndex = index;
          this.selectedIndex = index;
        })
        .onAnimationStart((index: number, targetIndex: number) => {
          if (index === targetIndex) {
            return;
          }
          // selectedIndex控制自定义TabBar内Image和Text颜色切换
          this.selectedIndex = targetIndex;
        })
        .width('100%')
        .height('100%');
      }
      .width('100%')
      .height('100%')
      .padding({top: '123px',bottom: '91px'})
    }
    .hideTitleBar(true)
    .height('100%')
    .width('100%')
    .mode(NavigationMode.Stack);
  }
}
```

NewWeb.ets：

```
import { webview } from '@kit.ArkWeb';

@Builder
export function NewWebBuilder() {
  NewWeb();
}

@Entry
@Component
export struct NewWeb {
  pageInfos: NavPathStack = new NavPathStack();
  @State url: string = '';
  controller: webview.WebviewController = new webview.WebviewController();

  build() {
    NavDestination() {
      Stack() {
        // 使用新的Web组件显示新页面
        Web({ src: $rawfile(this.url), controller: this.controller })
          .height('100%')
          .width('100%')
          .fileAccess(true)
          .geolocationAccess(false);
      }
      .height('100%')
      .width('100%')
      .padding({top: '123px',bottom: '91px'})
    }
    .onBackPressed(() => {
      this.pageInfos.pop(); // 弹出路由栈栈顶元素
      return true;
    })
    .onReady((context: NavDestinationContext) => {
      this.pageInfos = context.pathStack;
      this.url = this.pageInfos.getParamByName('newWeb').toString(); // 获取网页地址
    })
    .hideTitleBar(true)
    .height('100%')
    .width('100%')
    .backgroundColor(Color.White);
  }
}
```
