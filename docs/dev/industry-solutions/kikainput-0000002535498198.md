## 场景介绍

在输入法键盘上方根据按键内容展示对应的联想词框是开发一个输入法应用的高频使用场景之一。

本示例实现了两种常用场景：

1. 通过固定键盘高度来预留顶部联想词区域，实现在键盘区域内显示联想词框。
2. 采用全屏热区控制的方案，利用键盘窗口的[adjustPanelRect](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/js-apis-inputmethodengine#adjustpanelrect15)接口将键盘设置为全屏模式，通过控制热区，来改变键盘响应区域，实现在键盘区域外显示联想词框。

## 实现思路

**场景一**：如果需要实现键盘区域内显示联想词，可以通过固定键盘高度来预留顶部联想词区域。实现步骤如下：

1. 创建输入法面板。

   ```
   inputMethodAbility.createPanel(this.mContext, panelInfo).then((panel: inputMethodEngine.Panel) => {
     this.panel = panel;
     panel.resize(dWidth, keyHeight).then(() => {
       panel.moveTo(0, this.barPosition).then(() => {
         panel.setUiContent('InputMethodExtensionAbility/pages/Index').then(() => {
           this.inputHandle.addLog('loadContent finished');
         });
       });
     });
   });
   ```
2. 自定义联想词区域样式。

   ```
   import { KeyStyle, StyleConfiguration } from '../common/StyleConfiguration';

   @Preview
   @Component
   export struct SmallMenu {
     @Consume lianxiang: string = '';
     @StorageLink('inputStyle') inputStyle: KeyStyle = StyleConfiguration.getSavedInputStyle();

     build() {
       Row() {
         Row({ space: 50 }) {
           Text(this.lianxiang)
             .fontSize(25)
             .opacity(0.7)

           Text(this.lianxiang)
             .fontSize(25)
             .opacity(0.7)

           Text(this.lianxiang)
             .fontSize(25)
             .opacity(0.7)
         }
         .margin({ left: '16px', right: '16px' })
       }
       .width('100%')
       .height(this.inputStyle.downMenuHeight)
       .justifyContent(FlexAlign.Start)

     }
   }
   ```
3. 在主键盘页面中固定键盘高度来预留顶部联想词区域。

   ```
   import deviceInfo from '@ohos.deviceInfo';
   import Log from '../../model/Log';
   import { EditView } from '../../components/EditView';
   import { InputHandler } from '../model/KeyboardController';
   import {
     MenuType,
     SubMenuType
   } from '../../model/KeyboardKeyData';
   import { KeyMenu } from '../../components/KeyMenu';
   import { NumberMenu } from '../../components/NumberMenu';
   import { StyleConfiguration, KeyStyle } from '../../common/StyleConfiguration';
   import { SymbolMenu } from '../../components/SymbolMenu';
   import { Submenu } from '../../components/Submenu';
   import { TopMenu } from '../../components/TopMenu';
   import { SmallMenu } from '../../components/SmallMenu';
   import { inputMethodEngine } from '@kit.IMEKit';

   const DEVICE_TYPE: string = deviceInfo.deviceType;
   const TAG: string = 'index->';

   @Entry
   @Component
   struct Index {
     @Provide menuType: number = MenuType.NORMAL;
     @StorageLink('inputPattern') @Watch('inputPatternChange') inputPattern: InputType = InputType.Normal;
     @StorageLink('submenuType') submenuType: number = SubMenuType.NORMAL;
     @StorageLink('isLandscape') @Watch('change') isLandscape: boolean = false;
     @StorageLink('isRkDevice') isRkDevice: boolean = true;
     @StorageLink('inputStyle') inputStyle: KeyStyle =
       StyleConfiguration.getInputStyle(this.isLandscape, this.isRkDevice, DEVICE_TYPE);
     private panel: inputMethodEngine.Panel | undefined;
     @StorageLink('subtypeChange') subtypeChange: number = 0;
     @Provide lianxiang: string = '';

     aboutToAppear(): void {
       inputMethodEngine.getKeyboardDelegate().on('editorAttributeChanged', (attr: inputMethodEngine.EditorAttribute) => {
         console.info('recv editorAttributeChanged, immersiveMode: ', attr.immersiveMode);
         if (attr.immersiveMode == 1) {
           this.panel?.setImmersiveMode(inputMethodEngine.ImmersiveMode.DARK_IMMERSIVE);
           console.info('recv editorAttributeChanged, panel:', this.panel?.getImmersiveMode());
         }
       });
     }

     onBackPress(): boolean {
       Log.showInfo(TAG, 'kikaInput onBackPress');
       this.submenuType = SubMenuType.NORMAL;
       InputHandler.getInstance().hideKeyboardSelf();
       return true;
     }

     inputPatternChange(): void {
       if (this.inputPattern === InputType.Number || this.inputPattern === InputType.PhoneNumber) {
         this.menuType = MenuType.NUMBER;
       } else {
         this.menuType = MenuType.NORMAL;
       }
     }

     change(): void {
       AppStorage.set('inputStyle', StyleConfiguration.getInputStyle(this.isLandscape, this.isRkDevice, DEVICE_TYPE));
     }

     build() {
       Stack() {
         Column() {
           if (this.lianxiang.length === 0) {
             TopMenu()
           } else {
             SmallMenu() // 预留联想词区域
           }
           Column() {
             if (this.submenuType > SubMenuType.NORMAL) {
               if (this.submenuType === SubMenuType.MENU) {
                 Submenu()
               } else {
                 EditView();
               }
             } else {
               if (this.menuType === MenuType.NORMAL) {
                 if (this.subtypeChange == 0) {
                   KeyMenu()
                 } else {
                   NumberMenu()
                 }
               } else if (this.menuType === MenuType.NUMBER) {
                 NumberMenu()
               } else {
                 SymbolMenu()
               }
             }
           }
           .width('100%')
           .layoutWeight(1)
           .justifyContent(FlexAlign.Center)
         }
         .height('100%')
       }
       .height('100%')
     }
   }
   ```

**场景二**：如果需要实现键盘区域外显示联想词，由于系统限制，非系统级输入法应用无法在键盘区域之外创建或显示内容，但是可采用全屏热区控制的方案，利用键盘窗口的[adjustPanelRect](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/js-apis-inputmethodengine#adjustpanelrect15)接口将键盘设置为全屏模式，通过控制热区，来改变键盘响应区域，这样的话不受面板大小限制，可在任意位置自定义UI页面。实现步骤如下：

1. 定义横屏和竖屏状态下的热区。热区是一个矩形区域，包含left、top、width和height属性。各个属性的含义见[Rect](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-window-i#rect7)。
2. 定义横屏和竖屏状态下的避让区，并设置全屏模式。参数设置见[EnhancedPanelRect](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/js-apis-inputmethodengine#enhancedpanelrect15)。

```
inputMethodAbility.createPanel(this.mContext, panelInfo).then((panel: inputMethodEngine.Panel) => {
  this.panel = panel;
  let landscapeRect1: window.Rect = {
    left: 300,
    top: 650,
    width: 2000,
    height: 500
  };
  let landscapeInputRegion: Array<window.Rect> = [landscapeRect1];

  // 设置热区，此区域代表键盘操作区域，优先级最高
  let portraitRect1: window.Rect = {
    left: 0,
    top: 1613, // 距离键盘顶部位置
    width: 1200, // 矩形区域的宽度
    height: 1000 // 矩形区域的高度
  };
  let portraitInputRegion: Array<window.Rect> = [portraitRect1];

  let panelFlag: inputMethodEngine.PanelFlag = inputMethodEngine.PanelFlag.FLG_FIXED;

  // 设置键盘避让区
  let panelRect: inputMethodEngine.EnhancedPanelRect = {
    landscapeAvoidY: 650, // 横屏状态的避让区
    landscapeInputRegion: landscapeInputRegion,
    portraitAvoidY: 1613, // 竖屏状态的避让区，应用内其他系统组件会对避让线以下的输入法面板区域进行避让
    portraitInputRegion: portraitInputRegion,
    fullScreenMode: true
  };
  panel.adjustPanelRect(panelFlag, panelRect);
  panel.resize(100, 100).then(() => {
    panel.moveTo(0, this.barPosition).then(() => {
      panel.setUiContent('InputMethodExtensionAbility/pages/Index').then(() => {
        this.inputHandle.addLog('loadContent finished');
      });
    });
  });
});
```

## 约束与限制

* 本示例支持API Version 20 Release及以上版本。
* 本示例支持HarmonyOS 6.0.0 Release SDK及以上版本。
* 本示例需要使用DevEco Studio 6.0.0 Release及以上版本进行编译运行。
