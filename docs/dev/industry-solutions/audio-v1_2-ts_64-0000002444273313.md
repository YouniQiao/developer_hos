## 场景介绍

图文合成是影音娱乐类应用的高频使用场景之一，如图片编辑时，在图片上添加文字保存。

本示例使用[组件截图](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/js-apis-arkui-componentsnapshot)获得图片组件与文字组件的[PixelMap](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-image-pixelmap)，将PixelMap传递到Native侧使用[OpenGL ES](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/opengles)合成，生成添加文字内容的图片，并保存到相册。

## 实现思路

1. 通过[PhotoViewPicker](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-photoaccesshelper-photoviewpicker)拉起系统图库，让用户选择需要编辑的图片，在图片上添加文本框输入文字。
2. 通过[组件截图](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/js-apis-arkui-componentsnapshot)获得图片组件截图[PixelMap](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/arkts-apis-image-pixelmap)与图片上文本框组件截图PixelMap。并将读取两个PixelMap的像素数据传递到Native侧。

   ```
   imgPixelmap = this.getUIContext().getComponentSnapshot().getSync('img');
   txtPixelmap = this.getUIContext().getComponentSnapshot().getSync('txt');

   let imgInfo = await imgPixelmap.getImageInfo();
   let imgSize = imgInfo.size;

   let imgBuffer: ArrayBuffer = new ArrayBuffer(imgPixelmap.getPixelBytesNumber());
   await imgPixelmap.readPixelsToBuffer(imgBuffer);
   let txtBuffer: ArrayBuffer = new ArrayBuffer(txtPixelmap.getPixelBytesNumber());
   await txtPixelmap.readPixelsToBuffer(txtBuffer);
   ```
3. 使用[OpenGL ES](https://developer.huawei.com/consumer/cn/doc/harmonyos-references/opengles)离屏合成ArkTS侧传递的图片像素数据和文本像素数据。

   ```
   GLfloat verticesImg[] = {
         // First triangle
         1.0f, 1.0f, 0.0f, 1.0f, 0.0f,   // ...
         1.0f, -1.0f, 0.0f, 1.0f, 1.0f,  // ...
         -1.0f, -1.0f, 0.0f, 0.0f, 1.0f, // ...
         // Second triangle
         1.0f, 1.0f, 0.0f, 1.0f, 0.0f,   // ...
         -1.0f, -1.0f, 0.0f, 0.0f, 1.0f, // ...
         -1.0f, 1.0f, 0.0f, 0.0f, 0.0f,  // ...
      };
      GLuint vboImg;
      GLuint vaoImg;
      glGenVertexArrays(1, &vaoImg);
      glBindVertexArray(vaoImg);
      glGenBuffers(1, &vboImg);
      glBindBuffer(GL_ARRAY_BUFFER, vboImg);
      glBufferData(GL_ARRAY_BUFFER, sizeof(verticesImg), verticesImg, GL_STATIC_DRAW);
      const char *vertexShaderSource = R"(#version 300 es
         layout (location = 0) in vec4 vPosition;
         layout (location = 1) in vec2 vTexCoord;
         out vec2 texCoord;
         void main() {
               gl_Position = vPosition;
               texCoord = vTexCoord;
         }
      )";
      const char *fragmentShaderSource = R"(#version 300 es
         precision mediump float;
         out vec4 fragColor;
         in vec2 texCoord;
         vec4 color1;
         vec4 color2;
         uniform sampler2D texture1;
         uniform sampler2D texture2;
         void main() {
               color1 = texture(texture1, texCoord);
               color2 = texture(texture2, texCoord);
               if (color2.a == 0.0)  {
                  fragColor = color1;
               } else {
                  fragColor = color2;
               }
         }
      )";
      GLuint vertexShaderImg;
      vertexShaderImg = glCreateShader(GL_VERTEX_SHADER);
      glShaderSource(vertexShaderImg, 1, &vertexShaderSource, nullptr);
      glCompileShader(vertexShaderImg);
      GLuint fragmentShaderImg;
      fragmentShaderImg = glCreateShader(GL_FRAGMENT_SHADER);
      glShaderSource(fragmentShaderImg, 1, &fragmentShaderSource, nullptr);
      glCompileShader(fragmentShaderImg);
      GLuint shaderProgramImg;
      shaderProgramImg = glCreateProgram();
      glAttachShader(shaderProgramImg, vertexShaderImg);
      glAttachShader(shaderProgramImg, fragmentShaderImg);
      glLinkProgram(shaderProgramImg);
      glUseProgram(shaderProgramImg);

      GLuint textureId[2];
      glGenTextures(2, textureId);

      glBindTexture(GL_TEXTURE_2D, textureId[0]);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
      glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, imgData);
      glGenerateMipmap(GL_TEXTURE_2D);
      glUniform1i(glGetUniformLocation(shaderProgramImg, "texture1"), 0);

      glBindTexture(GL_TEXTURE_2D, textureId[1]);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
      glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
      glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, txtData);
      glGenerateMipmap(GL_TEXTURE_2D);
      glUniform1i(glGetUniformLocation(shaderProgramImg, "texture2"), 1);

      glActiveTexture(GL_TEXTURE0);
      glBindTexture(GL_TEXTURE_2D, textureId[0]);
      glActiveTexture(GL_TEXTURE1);
      glBindTexture(GL_TEXTURE_2D, textureId[1]);

      glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (GLvoid *)0);
      glEnableVertexAttribArray(0);
      glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 5 * sizeof(GLfloat), (GLvoid *)(3 * sizeof(GLfloat)));
      glEnableVertexAttribArray(1);

      glDrawArrays(GL_TRIANGLES, 0, 6);
   ```
4. 图片合成完成后，拷贝OpenGL ES的纹理像素数据，通过[ArrayBuffer](https://developer.huawei.com/consumer/cn/doc/harmonyos-guides/arraybuffer-object)返回ArkTS侧，并将合成文字的图片保存到系统图库。

## 约束与限制

* 本示例支持API Version 20 Release及以上版本。
* 本示例支持HarmonyOS 6.0.0 Release SDK及以上版本。
* 本示例需要使用DevEco Studio 6.0.0 Release及以上版本进行编译运行。
